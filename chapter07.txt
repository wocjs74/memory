package chapter7;

public class BAnkAccountPO {
	//메소드 확장 클래스
	//메소드 - 기능 클래스-데이터(변수) ,기능(메소드)
	int balance= 0; //예금 잔액
	//static-메모리에 올려서 사용 준비
	public int deposit(int amount) { //기능(메소드)
		balance+=amount;
		return balance;
		
	}
	public int withdraw(int amount) { 
		balance-=amount;
		return balance;
	}
	public int checkmabalance() { //기능(메소드)
		System.out.println("잔액 : "+balance);
		return balance;
	}
	public static void main(String[] args) { //클래스 = 틀 ,인스턴스 = 메모리 올려서 사용 준비
//인스턴튼 생성
 	BAnkAccountPO yoon=new BAnkAccountPO();
 	BAnkAccountPO p = new BAnkAccountPO();
	BAnkAccountPO park=new BAnkAccountPO();
	//입금
	yoon.deposit(5000);
	park.deposit(7000);
	p.deposit(3000);
	//잔액조회
	yoon.checkmabalance();
	p.checkmabalance();
	
	}
}
