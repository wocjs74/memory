package com.fxmx.simple;

public class Circle {
	double rad;
	final double PI;
	
	public Circle(double r ) {
		rad =r ;
		PI = 3.14;
		
	}
	public double getArea() {
		return (rad*rad)*PI;
	}
	public double getPerimeter() {
		return (rad*rad)*PI;
}
}
